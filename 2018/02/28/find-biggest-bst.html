아주 큰 트리에서 가장 큰 BST 찾기

February 28, 2018

  • 문제: You are given a very large binary tree, return the root node of the largest BST in the given tree
  • 왼쪽, 오른쪽 순환하면서 BST 조건에 맞는 노드를 찾기 시작한다
  • 왼쪽 노드

``` int find_biggest_bst(node* ptr, int& min, int &max, int& total_nodes, node*& head) {
if(!ptr) return 0;

int left = find_biggest_bst(ptr, min, max, total_nodes, head);
int current_min = (left ==0)?ptr->value: min;
if((left==-1) || (left != 0 && ptr->value > max)) {
    return -1;
}

int right = find_biggest_bst(ptr, min, max, total_nodes, head);
int current_max = (right == 0)? ptr->value:max;
if(right == -1 || (right != 0 && ptr->value < min)) {
    return -1;
}

int total = left + right + 1;
current_min = min;
current_max = max;

if(total_nodes < total) {
    total_ndoes = total;
}
return total; } '''
