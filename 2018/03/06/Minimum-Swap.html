Minimum Swap

March 6, 2018

Minimum Swap 문제

  • Word Ladder 문제와 유사
  • Minimum number of swaps of chars in only one string to make two strings the same

코드

struct node {
    string str;
    int steps;
    node(string s, int st):str(s), steps(st){}
};

int numberOfSteps(string& s1, const string& s2) {
    map<string,int> visited;
    int steps = 0 ;
    queue<node> que;
    que.push(node(s1,0));
    
    while(!que.empty()) {
        node n = que.front(); 
        que.pop();
        if(n.str == s2)
            return n.steps;
        
        for(int i = 0; i < n.str.size(); i++){
            for(int j = i+1; j < n.str.size(); j++) {
                string derived = n.str;
                if(derived[i] != derived[j]) {
                    swap(derived[i], derived[j]);
                    if(!visited.count(derived)) {
                        que.push(node(derived, n.steps+1));
                        visited.insert(make_pair(derived,1));
                    }
                }
            }
        }
    }
    return -1;
}
