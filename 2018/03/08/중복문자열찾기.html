문자열 안에서 중복 문자열 찾기

  • Give a string, finds all duplicate substrings of length k
set<string> findDups(string& input, int k)
    set<string> temp;
    set<string> result;
    
    for(int i = 0 ; i < input.size() - k, i++ ) {
        string str = input.substr(i,k);
        if(temp.count(str) !=0)
            result.insert(str)
        else
            temp.insert(str);
    }
    
    return result;
}
