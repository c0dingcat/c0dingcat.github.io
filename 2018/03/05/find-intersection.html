2개의 정렬된 Array에서 교차점 찾기

  • 그냥 무식하게 찾으면 O(m*n)
  • 아래 방식으로 하면 최대 O(m+n)
  • 하나씩 요소들을 검사해서 A[i] > B[j] , B[j] > A[i]할 때마다 인덱스 변경, 서로 값이 같을때까지 찾기
  • 각 배열 안에 있는 요소들의 크기를 비교해서 어느쪽의 인덱스를 증가시켜줄지를 결정한다
vector<int> findIntersection(vector<int> A, vector<int> B) {
  vector<int> intersection;
  int n1 = A.size();
  int n2 = B.size();
  int i = 0, j = 0;
  while (i < n1 && j < n2) {
    if (A[i] > B[j]) {
      j++;
    } else if (B[j] > A[i]) {
      i++;
    } else {
      intersection.push_back(A[i]);
      i++;
      j++;
    }
  }
  return intersection;
}
